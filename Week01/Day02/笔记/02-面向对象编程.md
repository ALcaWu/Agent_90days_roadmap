## 一、类与对象

### 1.1 类的定义

```python
class Agent:
    """Agent基类"""
    
    def __init__(self, name, model="gpt-4"):
        """构造函数"""
        self.name = name
        self.model = model
        self.history = []
    
    def chat(self, message):
        """发送消息"""
        self.history.append({"role": "user", "content": message})
        return f"[{self.name}] 收到: {message}"
    
    def get_history(self):
        """获取对话历史"""
        return self.history


# 创建实例
agent = Agent("助手A", "gpt-3.5-turbo")
print(agent.chat("你好"))
print(agent.get_history())
```

### 1.3 特殊方法（魔术方法）

```python
class TokenCounter:
    """Token计数器"""
    
    def __init__(self, tokens=0):
        self.tokens = tokens
    
    def __str__(self):
        return f"TokenCounter(tokens={self.tokens})"
    
    def __repr__(self):
        return f"TokenCounter({self.tokens})"
    
    def __add__(self, other):
        return TokenCounter(self.tokens + other.tokens)
    
    def __len__(self):
        return self.tokens


counter1 = TokenCounter(100)
counter2 = TokenCounter(50)

print(str(counter1))       # TokenCounter(tokens=100)
print(counter1 + counter2) # TokenCounter(tokens=150)
print(len(counter1))       # 100
```

## 二、继承与多态

### 2.1 基本继承

```python
class BaseAgent:
    """Agent基类"""
    
    def __init__(self, name):
        self.name = name
    
    def process(self, input_text):
        """处理输入（子类需重写）"""
        raise NotImplementedError("子类必须实现此方法")


class ChatAgent(BaseAgent):
    """对话Agent"""
    
    def __init__(self, name, model="gpt-4"):
        super().__init__(name)  # 调用父类构造函数
        self.model = model
    
    def process(self, input_text):
        return f"[ChatAgent:{self.name}] 回复: {input_text}"


class CodeAgent(BaseAgent):
    """代码Agent"""
    
    def process(self, input_text):
        return f"[CodeAgent:{self.name}] 生成代码: {input_text}"


# 使用
chat_agent = ChatAgent("助手", "gpt-4")
code_agent = CodeAgent("程序员")

print(chat_agent.process("你好"))    # [ChatAgent:助手] 回复: 你好
print(code_agent.process("排序算法")) # [CodeAgent:程序员] 生成代码: 排序算法
```

### 2.3 多重继承与Mixin

```python
class LoggerMixin:
    """日志混入类"""
    
    def log(self, message):
        print(f"[LOG] {self.__class__.__name__}: {message}")


class CacheMixin:
    """缓存混入类"""
    
    def __init__(self):
        self._cache = {}
    
    def get_cache(self, key):
        return self._cache.get(key)
    
    def set_cache(self, key, value):
        self._cache[key] = value


class SmartAgent(BaseAgent, LoggerMixin, CacheMixin):
    """智能Agent（多重继承）"""
    
    def __init__(self, name):
        super().__init__(name)
        CacheMixin.__init__(self)
    
    def process(self, input_text):
        # 检查缓存
        cached = self.get_cache(input_text)
        if cached:
            self.log(f"命中缓存: {input_text}")
            return cached
        
        # 处理并缓存
        result = f"[SmartAgent] {input_text}"
        self.set_cache(input_text, result)
        self.log(f"处理完成: {input_text}")
        return result


agent = SmartAgent("智能助手")
print(agent.process("hello"))
print(agent.process("hello"))  # 第二次命中缓存
```

## 三、装饰器

### 3.1 函数装饰器

```python
import time
from functools import wraps

def timer(func):
    """计时装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        elapsed = time.time() - start
        print(f"[{func.__name__}] 耗时: {elapsed:.3f}s")
        return result
    return wrapper


def retry(max_attempts=3):
    """重试装饰器（带参数）"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts - 1:
                        raise
                    print(f"重试 {attempt + 1}/{max_attempts}")
        return wrapper
    return decorator


@timer
@retry(max_attempts=2)
def call_api(prompt):
    """模拟API调用"""
    time.sleep(0.1)
    return f"响应: {prompt}"


result = call_api("你好")
```

### 3.2 类装饰器

```python
def singleton(cls):
    """单例装饰器"""
    instances = {}
    
    @wraps(cls)
    def get_instance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]
    
    return get_instance


@singleton
class APIClient:
    """API客户端（单例）"""
    
    def __init__(self, api_key):
        self.api_key = api_key
        print("初始化APIClient")


client1 = APIClient("key1")
client2 = APIClient("key2")
print(client1 is client2)  # True（同一个实例）
```

### 3.3 内置装饰器

```python
class Config:
    """配置类"""
    
    def __init__(self):
        self._temperature = 0.7
    
    @property
    def temperature(self):
        """getter"""
        return self._temperature
    
    @temperature.setter
    def temperature(self, value):
        """setter"""
        if not 0 <= value <= 2:
            raise ValueError("temperature必须在0-2之间")
        self._temperature = value
    
    @staticmethod
    def get_default():
        """静态方法"""
        return {"model": "gpt-4", "temperature": 0.7}
    
    @classmethod
    def from_env(cls):
        """类方法"""
        import os
        return cls()


config = Config()
print(config.temperature)  # 0.7
config.temperature = 0.5   # 通过setter设置
```

## 四、异常处理

### 4.1 基本异常处理

```python
def divide(a, b):
    try:
        result = a / b
    except ZeroDivisionError:
        print("除数不能为0")
        return None
    except TypeError as e:
        print(f"类型错误: {e}")
        return None
    else:
        print("计算成功")
        return result
    finally:
        print("执行完毕")


divide(10, 2)   # 成功
divide(10, 0)   # 除零错误
divide("10", 2) # 类型错误
```

### 4.2 自定义异常

```python
class AgentError(Exception):
    """Agent基础异常"""
    pass


class ModelNotFoundError(AgentError):
    """模型未找到"""
    def __init__(self, model):
        self.model = model
        super().__init__(f"模型不存在: {model}")


class TokenLimitError(AgentError):
    """Token超限"""
    def __init__(self, current, limit):
        self.current = current
        self.limit = limit
        super().__init__(f"Token超限: {current}/{limit}")


def validate_tokens(tokens, limit=4096):
    if tokens > limit:
        raise TokenLimitError(tokens, limit)
    return True


try:
    validate_tokens(5000)
except TokenLimitError as e:
    print(f"错误: {e}")
```

### 4.3 上下文管理器

```python
from contextlib import contextmanager

@contextmanager
def api_context(api_key):
    """API调用上下文"""
    print("初始化连接...")
    client = {"api_key": api_key, "connected": True}
    try:
        yield client
    finally:
        print("关闭连接...")
        client["connected"] = False


with api_context("sk-xxx") as client:
    print(f"使用API: {client['api_key']}")
```

```python
class ConfigManager:
    """配置管理器"""
    
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._config = {}
        return cls._instance
    
    def set(self, key, value):
        self._config[key] = value
    
    def get(self, key, default=None):
        return self._config.get(key, default)
    
    def load_from_file(self, filepath):
        import json
        with open(filepath, "r") as f:
            self._config.update(json.load(f))
```
